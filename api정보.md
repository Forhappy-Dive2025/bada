해루질

```json
{
  "set1": {
    "type": "현재 수온",
    "data": ["18","°C"]
  },
  "set2": {
    "type": "일몰일출",
    //-> 제목 표시 없음 
    "data": [
      "일몰",
      " 19:20"
    ] // 일몰혹은 일출제공,시간은 "hh:mm"
  },
  "set3": {
    "type": "기상",
    "data": [
      "구름많음", //하늘 예보 값으로 텍스트로 제공
      "23°C" //온도
    ]
  },
  "set4": {
    "type": "바람",
    "data": [
      "3.4",
      "SSE"
    ]
    //바람세기, 방향
  },
  "set5": {
    "type": "간조만조", 
    "data": [
      "간조까지 4시간 남았어요.", //(간조만조종류)까지 (현재기준남은시간값)시간 남았아요텍스트 만들어서 제공
      "2.1", //수위
      "↓", 
      "오전 01:24 기준" //요청한 시간으로 (오전/오후) hh:mm 기준 텍스트 만들어서 제공
    ]
  }
}
```
